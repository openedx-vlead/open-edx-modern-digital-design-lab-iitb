module mux4to1(select, d, q);

input[1:0] select;
input[3:0] d;
output     q;

wire       q, q1, q2, q3, q4, select0_bar, select1_bar;
wire[1:0]  select;
wire[3:0]  d;

not n1( select0_bar, select[0] );
not n2( select1_bar, select[1] );

and a1( q1, select0_bar, select1_bar, d[0]);
and a2( q2, select[0]  , select1_bar, d[1]);
and a3( q3, select0_bar, select[1]  , d[2]);
and a4( q4, select[0]  , select[1]  , d[3]);

or o1( q, q1, q2, q3, q4 );

endmodule
